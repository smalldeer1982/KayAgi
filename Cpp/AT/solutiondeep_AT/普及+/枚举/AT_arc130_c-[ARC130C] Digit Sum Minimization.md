inline pip calc(int a,int b){
    for(int i=1;i<=9;i++)ts[i]=cs[i],tt[i]=ct[i];

    for(int i=1;i<=9;i++)

            ts[i]-=c;tt[j]-=c;res+=c;
            S+=str(c,i+'0'),T+=str(c,j+'0');

    res+=max(ts[9],tt[9]);
    for(int i=9;i>=1;i--)
        S+=str(ts[i],i+'0'),T+=str(tt[i],i+'0');

}

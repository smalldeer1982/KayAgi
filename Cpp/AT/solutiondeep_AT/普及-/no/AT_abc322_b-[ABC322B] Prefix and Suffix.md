### 核心代码示例

```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
    int N, M;
    cin >> N >> M;
    string S, T;
    cin >> S >> T;

    bool isPrefix = (T.substr(0, N) == S);
    bool isSuffix = (T.substr(M - N, N) == S);

    if (isPrefix && isSuffix) {
        cout << 0 << endl;
    } else if (isPrefix) {
        cout << 1 << endl;
    } else if (isSuffix) {
        cout << 2 << endl;
    } else {
        cout << 3 << endl;
    }

    return 0;
}
```

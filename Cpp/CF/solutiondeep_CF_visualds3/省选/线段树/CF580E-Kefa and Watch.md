struct Node {
    int l, r, tag;
    ll h1, h2;
    void update(ll v) {
        tag = v;
        h1 = v * sp1[r-l] % mod1;
        h2 = v * sp2[r-l] % mod2;

void pushup(Node &p, Node l, Node r) {
    int len = r.r - r.l + 1;
    p.h1 = (l.h1 * pw1[len] % mod1 + r.h1) % mod1;
    p.h2 = (l.h2 * pw2[len] % mod2 + r.h2) % mod2;
}

void pushdown(int o) {
    if (tr[o].tag == -1) return;
    tr[o<<1].update(tr[o].tag);
    tr[o<<1|1].update(tr[o].tag);
    tr[o].tag = -1;
}

    tr[o] = {l, r, -1, 0, 0};

        return;

    build(o<<1, l, mid);
    build(o<<1|1, mid+1, r);

        tr[o].update(v);
        return;

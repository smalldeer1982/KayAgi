### 思路

按题意模拟即可，若 $a+b=c$ 则输出 ``Right!``，否则输出 ``Wrong!``。这里有个技巧，用 ``scanf`` 读入时可以把 ``+`` 和 ``=`` 直接读入，就不需要我们自己分离出数字了。

```cpp
#include <bits/stdc++.h>
#define il inline
namespace Fast_IO {
	template <typename T> il void read(T &x) {
		x = 0; int f = 0; char ch = getchar();
		while (!isdigit(ch)) f |= (ch == '-'), ch = getchar();
		while (isdigit(ch)) x = x * 10 + (ch - 48), ch = getchar();
		x = f ? -x : x;
	}
	template <typename T, typename ...Args>
	il void read(T &x, Args& ...args) {read(x), read(args...);}
	template <typename T> il void write(T x, char c = '\n') {
		if (x) {
			if (x < 0) x = -x, putchar('-');
			char a[30]; short l;
			for (l = 0; x; x /= 10) a[l ++] = x % 10 ^ 48;
			for (l --; l >= 0; l --) putchar(a[l]);
		} else putchar('0');
		putchar(c);
	}
} using namespace Fast_IO;
using namespace std;

signed main() {
	int T; read(T);
	while (T --) {
		int a, b, c; scanf("%d+%d=%d", &a, &b, &c);
		if (a + b == c) puts("Right!");
		else puts("Wrong!");
	}
	return 0;
}
```

采用快读的方法去输入，每次输入字符将存储位切换至下一位，输入数字将存储位 $\times 10$ 再加上这个数字，最后判断是否相等即可。

```cpp
#include<bits/stdc++.h>
using namespace std;
int t;
string s;
int main(){
	cin>>t;
	while(t--){
		cin>>s;
		int l=s.size(),a[3]={},k=0;
		for(int i=0;i<l;i++){
			if((s[i]>57)||(s[i]<48))k++;
			else{
				a[k]*=10;
				a[k]+=s[i]-48;
			}
		}
		if((a[0]+a[1])==a[2])cout<<"Right!\n";
		else cout<<"Wrong!\n";
	}
	return 0;
}
```

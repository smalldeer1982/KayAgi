    struct Edge { int to, cap, cost, rev; };

    int dis[N], preV[N], preE[N];
    
    void addEdge(int u, int v, int cap, int cost) {

            while (!q.empty()) {

                        dis[e.to] = dis[u] + e.cost;
                        preV[e.to] = u; preE[e.to] = i;

        return cost;

typedef long long ll;
const int N = 2e5 + 5;

struct BIT {
    ll c1[N], c2[N];
    void add(int x, ll k) {
        for (int i = x; i < N; i += i & -i) 
            c1[i] += k, c2[i] += x * k;
    }
    ll ask(int x) {
        ll s1 = 0, s2 = 0;
        for (int i = x; i; i -= i & -i) 
            s1 += c1[i], s2 += c2[i];
        return (x + 1) * s1 - s2;
    }
    void update(int l, int r, ll k) {
        if (l > r) return;
        add(l, k); add(r + 1, -k);
    }
    ll query(int l, int r) { 
        return ask(r) - ask(l - 1); 
    }
} Tr;

struct LCT {
    int ch[N][2], fa[N], last[N];
    bool isr(int x) { 
        return ch[fa[x]][0] != x && ch[fa[x]][1] != x; 
    }
    void push(int x, int k) { 
        if (x) last[x] = k; 
    }
    void rot(int x) {
        int y = fa[x], z = fa[y];
        bool k = (ch[y][1] == x);
        if (!isr(y)) ch[z][ch[z][1] == y] = x;
        fa[x] = z, fa[y] = x;
        if (ch[x][!k]) fa[ch[x][!k]] = y;
        ch[y][k] = ch[x][!k];
        ch[x][!k] = y;

        while (!isr(x)) {
            int y = fa[x];
            if (!isr(y)) 
                rot((ch[y][0] == x) ^ (ch[y][1] == y) ? x : y);
            rot(x);
        }

        for (int y = 0; x; x = par[x]) {
            splay(x);

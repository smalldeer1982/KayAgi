int qpow(int a, int b) {

namespace PolyC {
    using Poly = vector<int>;

    int rev[400000];
    void ntt(Poly &a, int len, int on) {
        for (int i = 0; i < len; i++) if (i < rev[i]) swap(a[i], a[rev[i]]);
        for (int i = 1; i < len; i <<= 1) {
            int wn = qpow(G, (mod - 1) / (i << 1));
            for (int j = 0; j < len; j += (i << 1)) {

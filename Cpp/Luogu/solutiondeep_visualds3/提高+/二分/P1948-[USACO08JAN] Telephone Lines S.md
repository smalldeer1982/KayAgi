    e[++tot] = {v, w, head[u]};

bool check(int mid) {
    memset(dist, 0x3f, sizeof dist);
    deque<int> dq;
    dist[1] = 0;

                dist[v] = dist[u] + w;
                w ? dq.push_back(v) : dq.push_front(v);
            }
        }

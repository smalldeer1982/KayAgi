struct Point {

double get_y(double x, const Point& p1, const Point& p2) {
    if (abs(p1.x - p2.x) < 1e-8) return p1.y;
    double t = (x - p1.x) / (p2.x - p1.x);

double get_y_high(double x, const vector<Point>& poly) {

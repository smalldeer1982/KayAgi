using namespace std;
const int N = 3e5 + 5;
vector<int> g[N];
int dp[N][3], mdia[N], up[N], dia[N], ans;

void dfs1(int u, int fa) {
    for (int v : g[u]) if (v != fa) {
        dfs1(v, u);
        if (dp[v][0] + 1 > dp[u][0]) {
            dp[u][2] = dp[u][1];
            dp[u][1] = dp[u][0];
            dp[u][0] = dp[v][0] + 1;
        } else if (dp[v][0] + 1 > dp[u][1]) {
            dp[u][2] = dp[u][1];
            dp[u][1] = dp[v][0] + 1;
        } else if (dp[v][0] + 1 > dp[u][2]) {
            dp[u][2] = dp[v][0] + 1;
        }
        mdia[u] = max(mdia[u], mdia[v]);
    }
    mdia[u] = max(mdia[u], dp[u][0] + dp[u][1]);
}

void dfs2(int u, int fa) {
    for (int v : g[u]) if (v != fa) {
        int h = (dp[v][0] + 1 == dp[u][0] ? dp[u][1] : dp[u][0]);
        int now = max({dia[u], h + up[u], dp[u][0] + dp[u][1]});
        dia[v] = now;
        up[v] = max(up[u], h) + 1;
        dfs2(v, u);
    }
    ans = max(ans, mdia[u] + dia[u]);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n; cin >> n;
    for (int i = 1, u, v; i < n; ++i) {
        cin >> u >> v;
        g[u].push_back(v);
        g[v].push_back(u);
    }
    dfs1(1, 0);
    dfs2(1, 0);
    cout << ans + 1;
    return 0;
}

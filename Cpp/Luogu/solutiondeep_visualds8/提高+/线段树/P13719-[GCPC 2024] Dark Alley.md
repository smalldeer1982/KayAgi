int64 pow_mod(int64 b, int64 p, int64 m) {
    int64 r = 1;
    while (p) {
        if (p & 1) r = r * b % m;
        b = b * b % m;
        p >>= 1;
    }
    return r;
}

struct SegTree {
    int64 tag[MAXN << 2];

    void add(int p, int l, int r, int ql, int qr, int64 v) {
        if (ql > qr) return;
        if (ql <= l && r <= qr) {
            (tag[p] += v) %= MOD;
            return;

        int mid = (l + r) >> 1;
        add(p << 1, l, mid, ql, qr, v);
        add(p << 1 | 1, mid + 1, r, ql, qr, v);

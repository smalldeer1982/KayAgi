        int mid = (l + r) >> 1;

    void push_up(int p) {
        if (t[p << 1].mn <= t[p << 1 | 1].mn) t[p] = t[p << 1];
        else t[p] = t[p << 1 | 1];

        t[rs].mn += t[p].tag; t[rs].tag += t[p].tag;
        t[p].tag = 0;

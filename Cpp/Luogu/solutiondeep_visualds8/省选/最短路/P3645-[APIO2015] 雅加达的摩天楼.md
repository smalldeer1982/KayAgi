    while (!q.empty()) {

        int nxt;
        nxt = pos - jump;
        if (nxt == T || nxt + jump == T) return cout << step + 1, 0;
        if (nxt >= 0 && !vis[nxt][jump]) {
            vis[nxt].set(jump);
            if (!visPos[nxt]) {
                visPos[nxt] = true;
                for (int p : doge[nxt]) if (!vis[nxt][p])
                    vis[nxt].set(p), q.emplace_front(nxt, p, step + 1);
            }
            q.emplace_back(nxt, jump, step + 1);
        }
        nxt = pos + jump;
        if (nxt < n && !vis[nxt][jump]) {
            vis[nxt].set(jump);
            if (!visPos[nxt]) {
                visPos[nxt] = true;
                for (int p : doge[nxt]) if (!vis[nxt][p])
                    vis[nxt].set(p), q.emplace_front(nxt, p, step + 1);
            }
            q.emplace_back(nxt, jump, step + 1);
        }
    }
    cout << -1;
    return 0;
}

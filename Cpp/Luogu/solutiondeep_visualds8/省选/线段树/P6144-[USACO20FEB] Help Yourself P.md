void build(int p, int l, int r) {

void push(int p) {

        tr[p << 1].val[i] = 1LL * tr[p << 1].val[i] * t % P;

void add(int p, int l, int r, int pos, int tmp[]) {

        fac = 1LL * fac * (i + 1) % P;
